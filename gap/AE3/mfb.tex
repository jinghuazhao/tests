\section{Introduction}

We present a unified framework for estimating genetic and environmental
variance components from \textbf{nuclear family trios}
(mother–father–child) using standard linear mixed models.

The approach follows the key idea of Rabe-Hesketh, Skrondal \& Gjessing (2008):
pedigree likelihoods can be written as mixed models by exploiting
covariance equivalence.

This allows biometric models to be fitted using standard software
(\texttt{nlme} in R) rather than specialised SEM packages.

\section{Quantitative Genetic Models}

\subsection{ACE model}

\[
y_{ij} = X_{ij}\beta + a_{ij} + c_j + e_{ij}
\]

\[
a_{ij}\sim N(0,\sigma_A^2),\qquad
c_j\sim N(0,\sigma_C^2),\qquad
e_{ij}\sim N(0,\sigma_E^2)
\]

\subsection{ADE model}

\[
y_{ij}=X_{ij}\beta + a_{ij} + d_{ij} + e_{ij}
\]

\[
a_{ij}\sim N(0,\sigma_A^2),\qquad
d_{ij}\sim N(0,\sigma_D^2),\qquad
e_{ij}\sim N(0,\sigma_E^2)
\]

Total phenotypic variance:

\[
\sigma_P^2=\sigma_A^2+\sigma_C^2+\sigma_D^2+\sigma_E^2
\]

Only a subset of these components can be identified from trio data.

\section{Pedigree Covariance in Nuclear Families}

The covariance between relatives $i$ and $k$ is

\[
\text{Cov}(y_i,y_k)=2\phi_{ik}\sigma_A^2+\sigma_C^2
\]

\subsection{Kinship Coefficients}

\begin{center}
\begin{tabular}{lcc}
\toprule
Pair & $2\phi$ & Covariance \\ \midrule
Mother--Father & 0 & $\sigma_C^2$ \\
Mother--Child  & $\tfrac12$ & $\tfrac12\sigma_A^2+\sigma_C^2$ \\
Father--Child  & $\tfrac12$ & $\tfrac12\sigma_A^2+\sigma_C^2$ \\
\bottomrule
\end{tabular}
\end{center}

\subsection{Dominance covariance in trios}

Dominance effects require two alleles identical by descent.
Parent--child pairs do not share dominance deviations, therefore

\[
\text{Cov}_D(y_i,y_k)=0 \quad \text{for all pairs in trios.}
\]

Dominance contributes only to individual variance:

\[
Var(y)=\sigma_A^2+\sigma_D^2+\sigma_E^2
\]

\section{Mixed Model Representation}

\[
y = X\beta + Z_A u_A + Z_C u_C + Z_D u_D + e
\]

\[
u_A\sim N(0,\sigma_A^2 I),\quad
u_C\sim N(0,\sigma_C^2 I),\quad
u_D\sim N(0,\sigma_D^2 I)
\]

\section{Trio Design Matrix Construction}

Dataset indicators:

\[
\text{var1 = mother},\quad
\text{var2 = father},\quad
\text{var3 = child}
\]

Additive genetic contribution:

\[
A_{ij}=0.5\,\text{Mother}+0.5\,\text{Father}+1\,\text{Child}
\]

Shared family environment:
\[
C_{ij}=1
\]

Dominance deviation (individual-level):
\[
D_{ij}=1
\]

\section{Identifiability}

With nuclear family trios:

\begin{center}
\begin{tabular}{lc}
\toprule
Model & Identifiable? \\
\midrule
AE  & Yes \\
ACE & Yes \\
ADE & Yes \\
ACDE & No \\
\bottomrule
\end{tabular}
\end{center}

Shared environment and dominance cannot be separated simultaneously,
so ACE and ADE are mutually exclusive.

\section{Variance Components}

\subsection{AE model}
\[
\Sigma=\sigma_A^2K+\sigma_E^2I
\qquad
h^2=\frac{\sigma_A^2}{\sigma_A^2+\sigma_E^2}
\]

\subsection{ACE model}
\[
\Sigma=\sigma_A^2K+\sigma_C^2J+\sigma_E^2I
\]

\[
h^2=\frac{\sigma_A^2}{\sigma_P^2},\qquad
c^2=\frac{\sigma_C^2}{\sigma_P^2}
\]

\subsection{ADE model}
\[
\Sigma=\sigma_A^2K+\sigma_D^2I+\sigma_E^2I
\]

\[
h^2=\frac{\sigma_A^2}{\sigma_P^2},\qquad
d^2=\frac{\sigma_D^2}{\sigma_P^2}
\]

\section{Likelihood Model Comparison}

\[
LR=2(\ell_{\text{full}}-\ell_{\text{reduced}})
\]

\section{R Implementation}

\subsection{Unified Trio Model Function (AE / ACE / ADE)}

\begin{verbatim}
trio_model <- function(model, data, type=c("AE","ACE","ADE"))
{
  require(nlme)
  type <- match.arg(type)

  data$Acoef <- 0.5*data$var1 + 0.5*data$var2 + 1*data$var3
  data$Ccoef <- 1
  data$Dcoef <- 1
  data$indid <- 1:nrow(data)

  if(type=="AE")
    rand <- list(familyid = pdDiag(~ Acoef - 1))

  if(type=="ACE")
    rand <- list(familyid = pdDiag(~ Acoef + Ccoef - 1))

  if(type=="ADE")
    rand <- list(
      familyid = pdDiag(~ Acoef - 1),
      indid    = pdDiag(~ Dcoef - 1)
    )

  fit <- lme(model, random=rand, data=data, method="ML",
             control=lmeControl(opt="optim"))

  pars <- attr(fit$apVar,"Pars")

  varE <- exp(2*pars[1])
  varA <- exp(2*pars[2])
  varC <- if(type=="ACE") exp(2*pars[3]) else 0
  varD <- if(type=="ADE") exp(2*pars[3]) else 0

  varP <- varA + varC + varD + varE

  list(
    fit = fit,
    var = c(A=varA,C=varC,D=varD,E=varE),
    h2  = varA/varP,
    c2  = varC/varP,
    d2  = varD/varP
  )
}
\end{verbatim}

\subsection{Model Comparison}

\begin{verbatim}
compare_trio_models <- function(model,data)
{
  AE  <- trio_model(model,data,"AE")
  ACE <- trio_model(model,data,"ACE")
  ADE <- trio_model(model,data,"ADE")

  tab <- anova(AE$fit, ACE$fit, ADE$fit)

  list(AE=AE, ACE=ACE, ADE=ADE, anova=tab)
}
\end{verbatim}

\section{Example}

\begin{verbatim}
library(gap.datasets)
data(mfblong)

model <- bwt ~ male + first + midage + highage + birthyr
res <- compare_trio_models(model, mfblong)

res$anova
res$ACE$h2
res$ADE$d2
\end{verbatim}

\section{Conclusion}

Nuclear family biometric models can be estimated entirely within linear
mixed models. Trio data allow estimation of AE, ACE or ADE models,
but shared environment and dominance cannot be estimated simultaneously.
